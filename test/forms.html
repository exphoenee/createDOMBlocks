createForm

Általáns form létrehozása tömböbe szedett input konfigurációkkal. Minden input típust támogat.

createForm({
  parent: "#app",
  id: "myForm",
  inputs: [
    { type: "text", id: "name", labelText: "Nev:", placeholder: "Teljes nev" },
    { type: "email", id: "email", labelText: "Email:" },
    { type: "select", id: "role", labelText: "Szerep:",
      options: [{ text: "Admin", value: "admin" }, { text: "User", value: "user" }] },
    { type: "textarea", id: "msg", labelText: "Uzenet:", rows: 3 },
    { type: "checkbox", id: "accept", labelText: "Elfogadom" },
    { type: "submit", id: "submit", value: "Kuldes" },
  ],
  onSubmit: () => console.log("Form submitted!"),
});
