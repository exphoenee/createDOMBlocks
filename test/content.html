Tartalom Komponensek

createCard

Kártya komponens címmel, testtel és lábléccel.

createCard({
  parent: "#app",
  id: "card1",
  title: "Kártya cím",
  body: { tag: "p", text: "Ez egy kártya test tartalma." },
  footer: { tag: "div", children: [
    { tag: "button", text: "Mégse" },
    { tag: "button", text: "Mentés" },
  ]},
});

createContainer

    { tag: "p", text: "Ez egy containeren belüli szöveg." },

  ],
});
